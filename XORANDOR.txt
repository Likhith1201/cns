#include <stdio.h>

int main() {
    char *str = "Hello world";

    printf("Original String: %s\n", str);

    printf("After AND with 127: ");
    for (int i = 0; str[i] != '\0'; i++) {
        char and_result = str[i] & 127;
        printf("%c", and_result);
    }
    printf("\n");

    printf("After XOR with 127: ");
    for (int i = 0; str[i] != '\0'; i++) {
        char xor_result = str[i] ^ 127;
        printf("%c", xor_result);
    }
    printf("\n");

    return 0;
}
